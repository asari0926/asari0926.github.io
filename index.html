// 初期設定 let playerNumber = 5; let fightCount = 0; // 敵の数字をランダムで生成 function generateEnemyNumber() { return Math.floor(Math.random() * 5) + 1; } // ゲームのロジック function fight() { let enemyNumber = generateEnemyNumber(); document.getElementById('enemy-number').textContent = enemyNumber; let message = ''; // 勝敗判定 if (playerNumber > enemyNumber) { playerNumber += enemyNumber; // 敵に勝ったら、プレイヤーの数字が増える message = `勝ちました！プレイヤーの数字が${enemyNumber}増えて、${playerNumber}になりました。`; } else if (playerNumber === enemyNumber) { message = `同じ数字だったので負けました。`; } else { message = `プレイヤーの数字が大きいので負けました。`; } // 勝利カウント fightCount++; if (playerNumber >= 100) { message = `ゲームクリア！プレイヤーは100以上になりました！`; document.getElementById('fight-button').disabled = true; } else if (fightCount >= 10) { message += ' 10回戦いました。ゲーム終了です！'; document.getElementById('fight-button').disabled = true; } // メッセージ更新 document.getElementById('player-number').textContent = playerNumber; document.getElementById('message').textContent = message; } // ボタンにイベントを設定 document.getElementById('fight-button').addEventListener('click', fight);
